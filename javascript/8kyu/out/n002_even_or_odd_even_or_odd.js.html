/**
* Judge param is even or odd.
*
*
* @param {number} num - judged num
*
* @return {String} when param is even return "Even",and param is odd return "Odd"
*/
function even_or_odd(num) {
  return (num % 2 === 0) ? "Even" : "Odd";
}

export default even_or_odd;
